def get_rotations(n):

    """Generate all rotations of a number."""
    
    s = str(n)

def is_circular_prime(n):

    """Check if all rotations of a number are prime."""
    
    rotations = get_rotations(n)
    
    return all(isprime(rot) for rot in rotations)
    

def find_circular_primes(limit):
